Resize image preserving aspect ratio and crop to fit within 250x150. Images smaller are not altered.

 
    def resize_image
      image.combine_options do |c|
        c.resize  '200x150^' if image_too_large?
        c.gravity 'northwest'
        c.crop    '200x150+0+0'
        c.repage.+
      end
    end
